        \question[5] 因式分解：$(x^2+x+1)(x^2+x+2)-12$
        \begin{solution}{0.5cm}
            \methodonly $(x^2+x-2)(x^2+x+5)$
        \end{solution}
        \vspace{3.5cm}

        \question[5] 因式分解：$12(x+y)^2+11(x+y)(x-y)+2(x-y)^2$
        \begin{solution}{0.5cm}
            \methodonly 设$x+y=m,x-y=n$，原式可化为
            \[
                \begin{aligned}
                     & \phantom{=}12m^2+11mn+2n^2 \\
                     & = (3m+2n)(4m+n)            \\
                     & = (5x+y)(5x+3y)
                \end{aligned}
            \]
        \end{solution}
        \vspace{3.5cm}

        \question[5] 因式分解：$na$
        \begin{solution}{0.5cm}
            \methodonly $na$
        \end{solution}
        \vspace{3.5cm}
