        \question[5] 已知$x-3y+4z=1,2x+y-2z=2$，化简$x^2-2xy-3y^2+2xz+10yz-8z^2$
        \begin{solution}{0.5cm}
            \methodonly 将$x^2-2xy-3y^2+2xz+10yz-8z^2$因式分解
            \[
                \begin{aligned}
                    &\phantom{=} x^2-2xy-3y^2+2xz+10yz-8z^2 \\ 
                    &=(x-3y+4z)(x+y-2z) \\ 
                    &=(x-3y+4z)(2x+y-2z-x) \\ 
                    &=2-x
                \end{aligned}
            \]
        \end{solution}
